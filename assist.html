Integração com Google Sheets

Configure o endpoint do Google Apps Script (GAS) para salvar os dados automaticamente na planilha. O GAS deve aceitar POST com JSON.

Cole aqui a URL gerada ao publicar o GAS como Web App.

1. Abra o Google Apps Script (script.google.com)
2. Crie um novo projeto e cole o código abaixo
3. Implante como Web App (Executar como: eu; Quem pode acessar: qualquer pessoa)
4. Copie a URL gerada e cole no campo acima

Código GAS completo — cole tudo no editor: 
// ─────────────────────────────────────────────
// PASSO 1 — Rode setupPlanilha() UMA VEZ
//   Menu: Executar > setupPlanilha
//   Isso cria as abas e cabeçalhos automaticamente.
// ─────────────────────────────────────────────

var CAMPOS_P1 = [
  'timestamp','orgao','responsavel','periodoInicio','periodoFim','dataApresentacao',
  'problemas','gestaoInfo','processos','limitacoes',
  'diretrizes','objetivos','focos',
  'digitalDesc','digitalNum','dadosDesc','dadosNum',
  'reunioesDesc','reunioesNum','confDesc','confNum',
  'kpiReunioes','kpiDocs','kpiForms','kpiParticipantes',
  'kpiDigital','kpiOutro','kpiContext',
  'eficienciaAntes','eficienciaDepois','transAntes','transDepois',
  'orgAntes','orgDepois','sintese'
];

var CAMPOS_P2 = [
  'timestamp','p2Processos','p2Melhorias','p2Ganhos',
  'p2Bases','p2Paineis','p2Stack',
  'p2DocsTec','p2Comunicacao',
  'p2Automacao','p2Fluxos',
  'p2BemFeito','p2NaoFuncionou','p2Ajustes',
  'p2Gargalos','p2LimitacoesInst',
  'p2Proximos','p2Expansoes'
];

function setupPlanilha() {
  var ss = SpreadsheetApp.getActiveSpreadsheet();

  var abas = [
    { nome: 'parte_1_essencial',    campos: CAMPOS_P1 },
    { nome: 'parte_2_complementar', campos: CAMPOS_P2 }
  ];

  abas.forEach(function(cfg) {
    var aba = ss.getSheetByName(cfg.nome);
    if (!aba) {
      aba = ss.insertSheet(cfg.nome);
    }
    if (aba.getLastRow() === 0) {
      aba.appendRow(cfg.campos);
      aba.getRange(1, 1, 1, cfg.campos.length)
        .setBackground('#0D2E5A')
        .setFontColor('#FFFFFF')
        .setFontWeight('bold');
      aba.setFrozenRows(1);
    }
  });

  SpreadsheetApp.getUi().alert(
    'Planilha configurada!\n\nAbas criadas:\n- parte_1_essencial\n- parte_2_complementar\n\nAgora publique este script como Web App e cole a URL no assistente.'
  );
}

// ─────────────────────────────────────────────
// PASSO 2 — Publish > Deploy as Web App
//   Execute as: Me
//   Who has access: Anyone
// ─────────────────────────────────────────────

function doGet(e) {
  return ContentService
    .createTextOutput(JSON.stringify({ ok: true, status: 'online' }))
    .setMimeType(ContentService.MimeType.JSON);
}

function doPost(e) {
  try {
    var raw = e.postData.contents;
    var data = JSON.parse(raw);
    var ss = SpreadsheetApp.getActiveSpreadsheet();

    var camposDef = data.aba === 'parte_2_complementar' ? CAMPOS_P2 : CAMPOS_P1;
    var aba = ss.getSheetByName(data.aba);

    if (!aba) {
      aba = ss.insertSheet(data.aba);
      aba.appendRow(camposDef);
      aba.getRange(1, 1, 1, camposDef.length)
        .setBackground('#0D2E5A')
        .setFontColor('#FFFFFF')
        .setFontWeight('bold');
      aba.setFrozenRows(1);
    }

    var linha = camposDef.map(function(col) {
      return data.campos[col] !== undefined ? data.campos[col] : '';
    });
    aba.appendRow(linha);

    return ContentService
      .createTextOutput(JSON.stringify({ ok: true }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch(err) {
    return ContentService
      .createTextOutput(JSON.stringify({ ok: false, erro: err.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
